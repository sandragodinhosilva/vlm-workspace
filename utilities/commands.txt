CLUSTER VIEW  (gpuq + friends — tools in ~/utilities/slurm/, shortcuts in ~/.bashrc)
-----------------------------------------------------------------------------------
# gpuq — pretty slurm queue + per-node GPU table; SSH-probes nvidia-smi to flag ROGUE GPUs
#        (◆ = busy per nvidia-smi but NO slurm job claims it → node is NOT really free).
#        Lives at ~/utilities/slurm/gpuq, symlinked into ~/.local/bin so it's a bare command.
gpuq                                  # full view: slurm + ssh rogue-probe (top squeue now has a USER col)
gpuq --procs                          # + per-node sub-line per GPU process: gpu# owner pid MODEL (extra ssh)
gpuq --no-probe                       # slurm-only, fast, no ssh (no rogue detection)
gpuq --extra dev01,dev02              # also probe non-slurm nodes
gpuq --timeout 5                      # bump ssh connect timeout (default 3s)
NO_COLOR=1 gpuq                       # disable color
gq                                    # alias: FORCE_COLOR=1 watch -c -n5 gpuq  (live; Ctrl+C to quit)

# freenode — just the truly-clean idle nodes (slurm idle AND no rogue ◆). Use BEFORE pinning.
freenode

# rogue_nodes — comma list of nodes busy off-scheduler RIGHT NOW (fallback worker-30,31 if gpuq down).
#   sftvlm / nodestart now exclude this LIVE set instead of a hardcoded 30,31 (auto-frees them when clean).
rogue_nodes

# srun into nodes
sftvlm                                # srun full node, --exclude "$(rogue_nodes)", job sft-vlm
nodestart myrun                       # same, custom job-name

# myjobs — your slurm jobs + last log line of each running job (PROGRESS, not just "RUNNING").
